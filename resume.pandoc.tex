\renewcommand{\cventry}[7][.25em]{%
  \cvitem[#1]{#2}{%
    {\bfseries#3}%
    \ifthenelse{\equal{#4}{}}{}{, {\slshape#4}}%
    \ifthenelse{\equal{#5}{}}{}{, #5}%
    \ifthenelse{\equal{#6}{}}{}{, #6}%
    .\strut%
    \newline{\small\strut #7 \par}}}

\renewcommand{\cvitem}[3][.25em]{%
  \begin{mycvlist}
    \item[#2] #3
    \end{mycvlist}
    \addvspace{#1}}
